function nextV8(x1, y1, x2, y2, duration){
    delayTime =(Math.floor(Math.random() * 2) + 4) * 1000;         //这里设置每个视频的观看时间为 10-12 秒之间

function lookback(x1, y1, x2, y2, duration){
    sleep((Math.floor(Math.random() * 1) + 1) * 1000);

    function getDatetime() {
        var now = new Date();
        var hh = now.getHours();
        var mm = now.getMinutes();
        var ss = now.getSeconds();
        var clock = ""
        if (hh < 10) clock += "0";
        clock += hh + ":";
        if (mm < 10) clock += '0';
        clock += mm + ":";
        if (ss < 10) clock += '0';
        clock += ss;
        return clock;
    }

function wifi() {

    importClass(java.lang.Integer);
    importClass(java.lang.System);
    importClass(android.os.Build);
    //importClass(android.os.Build.VERSION_CODES);
    importClass(java.net.HttpURLConnection);
    importClass(java.net.URL);
    importClass(java.net.InetSocketAddress);
    importClass(java.net.Proxy);
    var client = new OkHttpClient().newBuilder().proxy(Proxy.NO_PROXY).build();
    var urlStr = "http://quan.suning.com/getSysTime.do";
    var url = new URL(urlStr);
    urlConnection = (HttpURLConnection);
    url.openConnection(Proxy.NO_PROXY);
    // log(isWifiProxy())
    var IS_ICS_OR_LATER = Build.VERSION.SDK_INT >= Build.VERSION_CODES.ICE_CREAM_SANDWICH;
    var proxyAddress;
    var proxyPort;
    if (IS_ICS_OR_LATER) {
        proxyAddress = System.getProperty("http.proxyHost");
        var portStr = System.getProperty("http.proxyPort");
        proxyPort = Integer.parseInt((portStr != null ? portStr : "-1"));
    } else {
        proxyAddress = android.net.Proxy.getHost(this);
        proxyPort = android.net.Proxy.getPort(this);
    }
    return (!TextUtils.isEmpty(proxyAddress)) && (proxyPort != -1);
}

function vpn() {

        var list = java.util.Collections.list(niList);

            var Nane = list.get(i).getName();
            if ("tun0".equals(Nane) || "ppp0".equals(Nane)) return true;
        }
    }
    return false;
