100万数据批量插入优化

染染星河 2021-01-12 MyBatis-Plus

springboot整合mybatis-plus多数据源 JDBC批量处理操作Batch方法，需要在连接数据库的url加入 ?rewriteBatchedStatements=true 设置支持批量处理

url: jdbc:mysql://localhost:3306/userdb?useServerPrepStmts=false&rewriteBatchedStatements=true&serverTimezone=GMT%2B8&useUnicode=true&characterEncoding=UTF-8&useSSL=false

   dynamic:
     primary: userdb #设置默认的数据源或者数据源组,默认值即为master

# server

import com.baomidou.dynamic.datasource.annotation.DS;
import com.ranran.yeshen.serve.UserService;
import lombok.extern.slf4j.Slf4j;
import org.apache.ibatis.annotations.Mapper;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;
import com.baomidou.mybatisplus.extension.service.impl.ServiceImpl;
import com.ranran.yeshen.mapper.UserMapper;
import com.ranran.yeshen.dao.User;
import org.springframework.transaction.annotation.Transactional;

import javax.sql.DataSource;
import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.PreparedStatement;
import java.sql.SQLException;
import java.util.List;
import java.util.concurrent.CompletableFuture;

@Service
@DS("userdb")
@Slf4j
public class UserServiceImpl extends ServiceImpl<UserMapper, User> implements UserService {

    @Autowired
    private UserMapper userMapper;
    @Autowired
    private DataSource dataSource;
    @Override
   // @Transactional
    public boolean insertBigData(List<User> list) {
        long startTime = System.currentTimeMillis();
        String sql ="insert into user (username) values (?)";
        Connection connection =null;
        PreparedStatement statement = null;
        boolean flag =false;
        try {
            connection= dataSource.getConnection();
            //connection.setAutoCommit(false);  //关闭自动提交
            statement= connection.prepareStatement(sql);
            for( User user:list){
                statement.setString(1,user.getUsername());
                statement.addBatch();
            }
            statement.executeBatch();
            flag = true;
            //connection.commit();   //提交事务
            //connection.setAutoCommit(true);
            long endTime = System.currentTimeMillis();
            log.warn("批量插入数"+list.size()+"耗时"+(endTime-startTime)/1000.0+"秒");
        } catch (SQLException e) {
            e.printStackTrace();

        }

        return flag;

    }


}

# controller

 @RequestMapping("/insert")
    public ResponseResult insertBigData (@RequestParam int num){
        List<User> list = new ArrayList<>();
        for (int i = 0; i < num; i++) {
            User user = new User();
            user.setUsername("yeshen"+i);
            list.add(user);
        }

        boolean flag = userService.insertBigData(list);
        if (flag){
            return ResponseResult.Success(200,"success");
        }
        return ResponseResult.Error();

    }
