染染星河 2021-01-12 EasyExcel

# PageUtil

/**
 * 分页工具类
 */
public class PageUtil {

    /**
     * 分页查询
     *
     * @param pageNum 页码
     * @param pageSize  页距（条/页）
     * @param supplier  查询
     * @param <T>       结果集类型
     * @return 结果
     */
    public static <T> PageModel<T> query(Integer pageNum, Integer pageSize, Supplier<List<T>> supplier) {
        Page<T> page = PageHelper.startPage(pageNum, pageSize);
        List<T> list = supplier.get();
        return PageModel.<T>builder()
                .pageNum(pageNum)
                .pageSize(pageSize)
                .total((int) page.getTotal())
                .dataList(list)
                .build();
    }

    
    
    public static <T> PageInfo<T> pageInfo(Integer pageNum, Integer pageSize, Supplier<List<T>> supplier) {
        Page<T> page = PageHelper.startPage(pageNum, pageSize);
        List<T> list = supplier.get();
        PageInfo<T> pageInfo = new PageInfo<>(list);
        return pageInfo;
    }

    
}

@JsonInclude(JsonInclude.Include.NON_NULL)
@Slf4j
public class ResponseResult<T> {
    /**
     * 状态码
     */
    private Integer code;
    /**
     * 提示信息，如果有错误时，前端可以获取该字段进行提示
     */
    private String msg;
    /**
     * 查询到的结果数据，
     */
    private T data;


    public ResponseResult(Integer code, String msg) {
        this.code = code;
        this.msg = msg;
    }

    public ResponseResult(Integer code, T data) {
        this.code = code;
        this.data = data;
    }

    public Integer getCode() {
        return code;
    }

    public void setCode(Integer code) {
        this.code = code;
    }

    public String getMsg() {
        return msg;
    }

    public void setMsg(String msg) {
        this.msg = msg;
    }

    public T getData() {
        return data;
    }

    public void setData(T data) {
        this.data = data;
    }

    public ResponseResult(Integer code, String msg, T data) {
        this.code = code;
        this.msg = msg;
        this.data = data;
    }

    public ResponseResult() {

    }

    public static ResponseResult Error(){
        ResponseResult responseResult =new ResponseResult();
        responseResult.setCode(500);
        responseResult.setMsg("操作失败");
        log.error("操作失败");
        return responseResult;
    }

    public static ResponseResult Error(String msg){
        ResponseResult responseResult =new ResponseResult();
        responseResult.setCode(500);
        responseResult.setMsg(msg);
        log.error(msg);
        return responseResult;
    }

        ResponseResult responseResult =new ResponseResult();
        responseResult.setCode(code);

        log.error(msg);
        return responseResult;
    }

        ResponseResult responseResult =new ResponseResult();
        responseResult.setCode(code);
        responseResult.setMsg(msg);

import com.alibaba.excel.EasyExcel;
import com.alibaba.excel.ExcelWriter;
import com.alibaba.excel.write.metadata.WriteSheet;
import com.alibaba.excel.write.style.column.LongestMatchColumnWidthStyleStrategy;
import lombok.extern.slf4j.Slf4j;
import org.apache.poi.util.Removal;
import org.javaboy.vhr.model.bus.FleetManagement;
import org.javaboy.vhr.model.bus.MonCompany;
import org.javaboy.vhr.service.bus.MonCompanyService;
import org.javaboy.vhr.utils.PageModel;
import org.javaboy.vhr.utils.ResponseResult;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.util.ObjectUtils;
import org.springframework.web.bind.annotation.*;
import org.springframework.web.multipart.MultipartFile;

import javax.servlet.http.HttpServletResponse;
import java.io.IOException;
import java.io.OutputStream;
import java.net.URLEncoder;
import java.text.SimpleDateFormat;
import java.util.ArrayList;
import java.util.Date;
import java.util.List;

@RestController
@RequestMapping("/bus")
@Slf4j
public class BusController {

    @Autowired
    private MonCompanyService monCompanyService;

    /**
     * 可根据条件搜索导出
     * @param keyword
     * @param response
     */
    @GetMapping("/exprot")
    public void exprotExcel(@RequestParam(required = false) String keyword ,HttpServletResponse response) {


        long start = System.currentTimeMillis();
        OutputStream outputStream;
        String sheetName ="导出数据列表";
        String time = new SimpleDateFormat("yyyy-MM-dd-hh-mm-ss").format(new Date());
        StringBuffer bf = new StringBuffer();
        String fileName = bf.append(sheetName)
                .append(time)
                .append(".xlsx")
                .toString();

        try {
            outputStream = response.getOutputStream();

            //添加响应头信息
            response.setHeader("Content-disposition", "attachment; filename=" + URLEncoder.encode(fileName, "UTF-8"));
            //设置类型
            response.setContentType("application/vnd.ms-excel;charset=UTF-8");
            //设置头
            response.setHeader("Pragma", "No-cache");
            //设置头
            response.setHeader("Cache-Control", "no-cache");
            //设置日期头
            response.setDateHeader("Expires", 0);

            PageModel<FleetManagement> companyPage = monCompanyService.getFleetManagementPage(1, 10, keyword);
            //获取总数据量
            Integer count = companyPage.getTotal();
            //每页多少个
            int maxSize=60000;   
            //导出Excel
            //如果总数据量多余10万，分页导出
            if(count>maxSize){
                //必须放到循环外，否则会刷新流
                ExcelWriter excelWriter=EasyExcel.write(outputStream).build();
                for (int i=0;i<(count/maxSize)+1;i++){
                    Integer pageNum = i+1;
                    Integer pageSize = maxSize;
                    //查询数据库  分页查询
                    PageModel<FleetManagement> pageModel = monCompanyService.getFleetManagementPage(pageNum, pageSize, keyword);
                    //获取datalist
                    List<FleetManagement> dataList = pageModel.getDataList();

                    WriteSheet writeSheet = EasyExcel.writerSheet(i, sheetName + (i + 1)+"页").head(FleetManagement.class)
                            .registerWriteHandler(new LongestMatchColumnWidthStyleStrategy()).build();
                    excelWriter.write(dataList,writeSheet);
                }

                //刷新流
                excelWriter.finish();
            }else {
                // 数据库中的数据小于 60000条
                Integer pageNum = 1;
                Integer pageSize = maxSize;
                PageModel<FleetManagement> pageModel = monCompanyService.getFleetManagementPage(pageNum, pageSize, keyword);
                List<FleetManagement> dataList = pageModel.getDataList();
                EasyExcel.write(outputStream,FleetManagement.class).registerWriteHandler(new LongestMatchColumnWidthStyleStrategy()).sheet(sheetName).doWrite(dataList);
            }
            outputStream.flush();
            response.getOutputStream().close();
            long end = System.currentTimeMillis();
            log.info("需要的时间===" + (end - start));
        } catch (Exception e) {
            log.error("导出失败：{}", e.getMessage());
            e.printStackTrace();
        }
    }

}
