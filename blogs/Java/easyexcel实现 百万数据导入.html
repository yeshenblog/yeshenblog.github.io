# ExcelUtils

// 有个很重要的点 ExcelListener 不能被spring管理，要每次读取excel都要new,然后里面用到spring可以构造方法传进去
public class ExcelUtils extends AnalysisEventListener<FleetManagement> {

    private List<FleetManagement> list = new ArrayList<>();
    /**
     * 每隔100w条存储数据库，实际使用中可以100w条，然后清理list ，方便内存回收
     */
    private static final int BATCH_COUNT =1000000;
    /**
     * 假设这个是一个DAO，当然有业务逻辑这个也可以是一个service。当然如果不用存储这个对象没用。
     */
    private MonCompanyService monCompanyService;

    /**
     * 如果使用了spring,请使用这个构造方法。每次创建Listener的时候需要把spring管理的类传进来
     */
    public ExcelUtils(MonCompanyService monCompanyService) {
        this.monCompanyService = monCompanyService;
    }
    /**
     * 这个每一条数据解析都会来调用
     */
    @Override
    public void invoke(FleetManagement fleetManagement, AnalysisContext analysisContext) {
//        System.out.println("解析到一条数据:========================"+fleetManagement.toString());
        // 数据存储到datas，供批量处理，或后续自己业务逻辑处理。

        if(!ObjectUtils.isEmpty(fleetManagement) ) {
             list.add(fleetManagement);
         }
        // 达到BATCH_COUNT了，需要去存储一次数据库，防止数据几万条数据在内存，容易OOM
        if(list.size() >= BATCH_COUNT){
            saveData();
            // 存储完成清理datas
             long runCount = analysisContext.readRowHolder().getRowIndex();
            //表的总行数
            long totalCount = analysisContext.readSheetHolder().getApproximateTotalRowNumber();
            list.clear();
        }
    }
    /**
     * 所有数据解析完成了 都会来调用
     */
    @Override
    public void doAfterAllAnalysed(AnalysisContext analysisContext) {
        saveData();//确保所有数据都能入库
    }
    /**
     * 加上存储数据库
     */
    private void saveData() {
        System.out.println("=============================="+list.size()+"条数据，开始存储到数据库");
        monCompanyService.insertFleet(list);

    }
}

@Data
@TableName(value = "fleet_management")
public class FleetManagement {

    @TableId(value = "company_id", type = IdType.AUTO)
    @ExcelIgnore
    private Integer companyId;

    @ExcelProperty("车队名称")
    @TableField(value = "company_name")
    private String companyName;

    @ExcelProperty("所属机构")
    @TableField(value = "agent_name")
    private String agentName;

    @ExcelProperty("所在地区")
    @TableField(value = "address")
    private String address;

    @ExcelProperty("安全负责人")
    @TableField(value = "contract")
    private String contract;

    @ExcelProperty("联系电话")
    @TableField(value = "phone")
    private String phone;

    @ExcelIgnore
    @TableField(value = "state")
    private Integer state;

}

# ServiceImpl

@Service
@Slf4j
public class MonCompanyServiceImpl extends ServiceImpl<MonCompanyMapper, MonCompany> implements MonCompanyService {

  @Autowired
  private MonCompanyMapper monCompanyMapper;

  @Autowired
  private DataSource dataSource;

  @Override
  // @Transactional
  public boolean insertBigData(List<MonCompany> list) {
    long startTime = System.currentTimeMillis();
    String sql ="insert into mon_company (agent_id,company_name,address,contract1,contract1_tel) values (?,?,?,?,?)";
    Connection connection =null;
    PreparedStatement statement = null;
    boolean flag =false;
    try {
      connection= dataSource.getConnection();
      //connection.setAutoCommit(false);  //关闭自动提交
      statement= connection.prepareStatement(sql);
      for( MonCompany monCompany:list){
        statement.setInt(1,monCompany.getAgentId());
        statement.setString(2,monCompany.getCompanyName());
        statement.setString(3,monCompany.getAddress());
        statement.setString(4,monCompany.getContract1());
        statement.setString(5,monCompany.getContract1Tel());
        statement.addBatch();
      }
      statement.executeBatch();
      flag = true;
      //connection.commit();   //提交事务
      //connection.setAutoCommit(true);
      long endTime = System.currentTimeMillis();
      log.warn("批量插入数"+list.size()+"耗时"+(endTime-startTime)/1000.0+"秒");
    } catch (SQLException e) {
      e.printStackTrace();

    }

    return flag;

  }


}
